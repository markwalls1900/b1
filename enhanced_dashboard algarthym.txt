        function runAlgorithmicAnalysis() {
            const companyType = document.getElementById('companyType').value;
            const investmentSize = document.getElementById('investmentSize').value;
            const riskTolerance = document.getElementById('riskTolerance').value;
            
            const resultsDiv = document.getElementById('algorithmResults');
            resultsDiv.innerHTML = '<div class="alert alert-info">Running algorithmic analysis...</div>';

            setTimeout(() => {
                const analysis = generateAlgorithmicAnalysis(companyType, investmentSize, riskTolerance);
                
                resultsDiv.innerHTML = `
                    <div class="alert alert-success">
                        <h6><i class="fas fa-check-circle me-2"></i>Algorithmic Analysis Complete</h6>
                    </div>
                    
                    <div class="row">
                        <div class="col-12 mb-3">
                            <div class="card">
                                <div class="card-body">
                                    <div class="row">
                                        <div class="col-8">
                                            <h6><i class="fas fa-calculator me-2"></i>Composite Score: ${analysis.composite_score}%</h6>
                                            <small class="text-muted">${analysis.tier_level} • ${analysis.confidence_level} confidence</small>
                                            <div class="mt-2">
                                                <span class="badge bg-success me-1">${analysis.roi_projection}% ROI</span>
                                                <span class="badge bg-primary me-1">$${analysis.cost_savings}M Savings</span>
                                                <span class="badge bg-${analysis.risk_level === 'Low Risk' ? 'success' : analysis.risk_level === 'Medium Risk' ? 'warning' : 'danger'}">${analysis.risk_level}</span>
                                            </div>
                                        </div>
                                        <div class="col-4 text-end">
                                            <div class="algorithm-score">${analysis.composite_score}%</div>
                                        </div>
                                    </div>
                                </div>
                            </div>
                        </div>
                    </div>
                `;
            }, 2000);
        }

        function generateAlgorithmicAnalysis(companyType, investmentSize, riskTolerance) {
            // Simulate algorithmic analysis
            const baseScore = 85;
            const companyMultiplier = companyType === 'tech' ? 1.1 : companyType === 'manufacturing' ? 1.05 : 1.0;
            const sizeMultiplier = investmentSize === 'large' ? 1.1 : investmentSize === 'enterprise' ? 1.15 : 0.95;
            const riskMultiplier = riskTolerance === 'low' ? 0.9 : riskTolerance === 'high' ? 1.1 : 1.0;
            
            const compositeScore = Math.round(baseScore * companyMultiplier * sizeMultiplier * riskMultiplier);
            
            let tierLevel, roiProjection, costSavings, riskLevel;
            
            if (compositeScore >= 85) {
                tierLevel = "TIER_1";
                roiProjection = 18.5;
                costSavings = 45;
                riskLevel = "Low Risk";
            } else if (compositeScore >= 70) {
                tierLevel = "TIER_2";
                roiProjection = 15.2;
                costSavings = 32;
                riskLevel = "Medium Risk";
            } else {
                tierLevel = "TIER_3";
                roiProjection = 22.8;
                costSavings = 28;
                riskLevel = "High Risk";
            }
            
            return {
                composite_score: compositeScore,
                tier_level: tierLevel,
                confidence_level: compositeScore > 85 ? "Very High (95%)" : compositeScore > 70 ? "High (85%)" : "Medium (75%)",
                roi_projection: roiProjection,
                cost_savings: costSavings,
                risk_level: riskLevel
            };
        }

        function showSuccess(message) {
            const alertDiv = document.createElement('div');
            alertDiv.className = 'alert alert-success alert-dismissible fade show';
            alertDiv.innerHTML = `
                ${message}
                <button type="button" class="btn-close" data-bs-dismiss="alert"></button>
            `;
            
            const container = document.querySelector('.container-fluid');
            container.insertBefore(alertDiv, container.firstChild);
            
            setTimeout(() => {
                if (alertDiv.parentNode) {
                    alertDiv.remove();
                }
            }, 5000);
        }
